import {Suspense} from 'react';
import {Await, useAsyncValue} from 'react-router';
import {
  type CartViewPayload,
  useAnalytics,
  useOptimisticCart,
} from '@shopify/hydrogen';
import type {HeaderQuery, CartApiQueryFragment} from 'storefrontapi.generated';
import {useAside} from '~/components/NativeComponents/Aside/Aside';
import navbarStyles from './Navbar.module.css';
import { activeLinkStyle} from '../utilities';
import { Link } from '../Link';
import { CountrySelector } from '~/components/CountrySelector';
import { Menu } from 'lucide-react';


interface HeaderProps {
  header: HeaderQuery;
  cart: Promise<CartApiQueryFragment | null>;
  isLoggedIn: Promise<boolean>;
  publicStoreDomain: string;

}




type Viewport = 'desktop' | 'mobile';

export function DefaultNavbar({
  header,
  isLoggedIn,
  cart,
  publicStoreDomain,
}: HeaderProps) {
  const {shop, menu} = header;
  return (
    <header className={`${navbarStyles.NavbarMainBackground} px-(--paddingX) z-101 sticky top-0 w-full h-15  flex flex-row items-center`} >

    <NavbarMenu
      shop={shop.name}
      menu={menu}
      viewport="desktop"
      primaryDomainUrl={header.shop.primaryDomain.url}
      publicStoreDomain={publicStoreDomain}
      />
      <NavbarCtas isLoggedIn={isLoggedIn} cart={cart} />
    </header>
  );
}

export function NavbarMenu({
  menu,
  primaryDomainUrl,
  publicStoreDomain,
  shop,
}: {
  shop:string;
  menu: HeaderProps['header']['menu'];
  primaryDomainUrl: HeaderProps['header']['shop']['primaryDomain']['url'];
  viewport: Viewport;
  publicStoreDomain: HeaderProps['publicStoreDomain'];
}) {


  const {close} = useAside();

  return (
    <nav className={` space-x-(--spacingX)  w-full flex flex-row items-center`}   role="navigation">

      <Link prefetch="intent" to="/" end>
        <span> {shop}</span>
      </Link>


        {(menu || FALLBACK_HEADER_MENU).items.map((item) => {
          if (!item.url) return null;
  
          // if the url is internal, we strip the domain
          const url =
            item.url.includes('myshopify.com') ||
            item.url.includes(publicStoreDomain) ||
            item.url.includes(primaryDomainUrl)
              ? new URL(item.url).pathname
              : item.url;
          return (
            <Link variant="nav"
              className={`md:flex hidden`}
              end
              key={item.id}
              onClick={close}
              prefetch="intent"
              style={activeLinkStyle}
              to={url}
            >
              {item.title}
            </Link>
          );
        })}



      </nav>
      
  );
}

function NavbarCtas({
  isLoggedIn,
  cart,
}: Pick<HeaderProps, 'isLoggedIn' | 'cart'>){
  return (

      <nav className={`items-center w-full justify-end space-x-(--spacingX) flex flex-row`} role="navigation">
        

        <NavbarMenuMobileToggle />


         <CountrySelector />
        <Link variant="nav" prefetch="intent" to="/account" style={activeLinkStyle}>
          <Suspense fallback="Sign in">
            <Await resolve={isLoggedIn} errorElement="Sign in">
              {(isLoggedIn) => (isLoggedIn ? 'Account' : 'Sign in')}
            </Await>
          </Suspense>
        </Link>
        <SearchToggle />
        <CartToggle cart={cart} />
      </nav>


  );
}

function NavbarMenuMobileToggle() {
  const {open} = useAside();
  return (
    <button
      className={`flex md:hidden`}
      onClick={() => open('mobile')}
    >
      <h3><Menu /></h3>
    </button>
  );
}

function SearchToggle() {
  const {open} = useAside();
  return (
    <button className={`${navbarStyles.reset}`} onClick={() => open('search')}>
      Search
    </button>
  );
}

function CartBadge({count}: {count: number | null}) {
  const {open} = useAside();
  const {publish, shop, cart, prevCart} = useAnalytics();

  return (
    <a
      className={navbarStyles.headerNavTitle}
      href="/cart"
      onClick={(e) => {
        e.preventDefault();
        open('cart');
        publish('cart_viewed', {
          cart,
          prevCart,
          shop,
          url: window.location.href || '',
        } as CartViewPayload);
      }}
    >
      Cart {count === null ? <span>&nbsp;</span> : count}
    </a>
  );
}

function CartToggle({cart}: Pick<HeaderProps, 'cart'>) {
  return (
    <Suspense fallback={<CartBadge count={null} />}>
      <Await resolve={cart}>
        <CartBanner />
      </Await>
    </Suspense>
  );
}

function CartBanner() {
  const originalCart = useAsyncValue() as CartApiQueryFragment | null;
  const cart = useOptimisticCart(originalCart);
  return <CartBadge count={cart?.totalQuantity ?? 0} />;
}
